<script>
    class CaptionGenerator {
        constructor() {
            this.form = document.getElementById('captionForm');
            this.loading = document.getElementById('loading');
            this.results = document.getElementById('results');
            this.captionsContainer = document.getElementById('captionsContainer');
            this.usageCount = this.getUsageCount();
            this.maxFreeUses = 5;
            
            this.updateUsageDisplay();
            this.setupEmailPopup();
            this.init();
        }

        init() {
            this.form.addEventListener('submit', this.handleSubmit.bind(this));
        }

        getUsageCount() {
            const saved = localStorage.getItem('captionUsage');
            if (saved) {
                const data = JSON.parse(saved);
                const today = new Date().toDateString();
                if (data.date === today) {
                    return data.count;
                }
            }
            return 0;
        }

        saveUsageCount(count) {
            const data = {
                date: new Date().toDateString(),
                count: count
            };
            localStorage.setItem('captionUsage', JSON.stringify(data));
        }

        updateUsageDisplay() {
            const remaining = Math.max(0, this.maxFreeUses - this.usageCount);
            document.getElementById('usageCount').textContent = remaining + '/' + this.maxFreeUses;
            
            if (remaining === 0) {
                document.getElementById('generateBtn').textContent = 'üîí Daily limit reached - Get unlimited access!';
                document.getElementById('generateBtn').onclick = function() { showEmailPopup(); };
            }
        }

        setupEmailPopup() {
            setTimeout(() => {
                if (this.usageCount >= 2) {
                    showEmailPopup();
                }
            }, 30000);
        }

        async handleSubmit(e) {
            e.preventDefault();
            
            if (this.usageCount >= this.maxFreeUses) {
                showEmailPopup();
                return;
            }
            
            const formData = new FormData(this.form);
            const data = {
                topic: formData.get('topic'),
                platform: formData.get('platform'),
                tone: formData.get('tone'),
                hashtags: formData.get('hashtags'),
                length: formData.get('length')
            };

            this.showLoading(true);
            await new Promise(resolve => setTimeout(resolve, 2000));
            
            const captions = this.generateCaptions(data);
            this.displayResults(captions);
            
            this.usageCount++;
            this.saveUsageCount(this.usageCount);
            this.updateUsageDisplay();
            this.showLoading(false);

            if (this.usageCount >= 3) {
                setTimeout(() => showEmailPopup(), 5000);
            }
        }

        generateCaptions(data) {
            const templates = {
                instagram: {
                    professional: [
                        data.topic + " üíº\\n\\nSharing insights on " + data.topic + ". What are your thoughts? Let me know in the comments below.\\n\\n{hashtags}",

                        data.topic + " - a game changer in today's market. üìä\\n\\nSwipe to learn more about why this matters.\\n\\n{hashtags}"
                    ],
                    casual: [
                        "Just had to share this! " + data.topic + " üòç\\n\\nHonestly, this made my day. Who else loves this?\\n\\n{hashtags}",

                        "Can we talk about " + data.topic + "? üó£Ô∏è\\n\\nI'm obsessed and I'm not sorry about it! Drop a ‚ù§Ô∏è if you agree.\\n\\n{hashtags}"
                    ],
                    funny: [
                        "Me: I don't need " + data.topic + "\\nAlso me: *gets " + data.topic + "* üòÇ\\n\\nWhy am I like this? Anyone else relate?\\n\\n{hashtags}",
                        data.topic + " hit different today ü§™\\n\\nI'm not sure what I expected, but it wasn't this. Send help! üòÖ\\n\\n{hashtags}",
                        "Plot twist: " + data.topic + " was actually worth the hype üé≠\\n\\nColor me surprised! What surprised you recently?\\n\\n{hashtags}"
                    ],
                    inspirational: [
                        "Every journey starts with a single step. üåü\\n\\n" + data.topic + " reminded me that progress isn't always linear, but it's always worth it.\\n\\n{hashtags}",
                        "Grateful for " + data.topic + " today üôè\\n\\nSometimes the universe gives us exactly what we need. Trust the process.\\n\\n{hashtags}",
                        "Your only limit is your mindset. üí≠\\n\\n" + data.topic + " taught me that challenges are just opportunities in disguise.\\n\\n{hashtags}"
                    ],
                    sales: [
                        "üî• Special offer alert! " + data.topic + "\\n\\nLimited time only - don't miss out! Link in bio for details.\\n\\n{hashtags}",
                        "Why choose " + data.topic + "? Let me tell you... ‚ú®\\n\\n‚úÖ Quality guaranteed\\n‚úÖ Fast delivery\\n‚úÖ Happy customers\\n\\nReady to try it? DM us!\\n\\n{hashtags}",
                        "Customer favorite: " + data.topic + " üèÜ\\n\\n\"Best purchase I've made this year!\" - Sarah M.\\n\\nWhat are you waiting for?\\n\\n{hashtags}"
                    ],
                    storytelling: [
                        "A few months ago, I discovered " + data.topic + "...\\n\\nLittle did I know it would completely change my perspective. Here's what happened next... üëá\\n\\n{hashtags}",
                        "Picture this: It's 6 AM and I'm thinking about " + data.topic + " üåÖ\\n\\nSounds crazy? Maybe. But let me tell you why this matters...\\n\\n{hashtags}",
                        "The story behind " + data.topic + " is incredible üìñ\\n\\nIt started with a simple idea and grew into something amazing. Swipe to see the journey...\\n\\n{hashtags}"
                    ]

            const hashtags = ['#motivation', '#inspiration', '#success', '#lifestyle', '#goals'];
            const platform = data.platform || 'instagram';
            const tone = data.tone || 'casual';
            const platformTemplates = templates[platform][tone] || templates.instagram.casual;
            
            const captions = [];
            const selectedTemplates = platformTemplates.slice(0, 3);
            
            selectedTemplates.forEach((template, index) => {
                let caption = template;
                
                if (data.hashtags === 'yes') {
                    const hashtagSet = hashtags.slice(0, 5);
                    caption = caption.replace('{hashtags}', hashtagSet.join(' '));
                } else {
                    caption = caption.replace('{hashtags}', '');
                }

                captions.push({
                    id: index + 1,
                    text: caption.trim(),
                    platform: platform,
                    tone: tone
                });
            });

            return captions;
        }

        displayResults(captions) {
            this.captionsContainer.innerHTML = '';
            
            captions.forEach(caption => {
                const captionEl = document.createElement('div');
                captionEl.className = 'caption-result';
                captionEl.innerHTML = '<h4>Version ' + caption.id + ' - ' + caption.tone.charAt(0).toUpperCase() + caption.tone.slice(1) + ' Style</h4><div class="caption-text">' + caption.text.replace(/\\n/g, '<br>') + '</div><button class="copy-btn" onclick="copyCaption(this, \'' + caption.text.replace(/'/g, "\\'") + '\')">üìã Copy Caption</button>';
                this.captionsContainer.appendChild(captionEl);
            });

            this.results.classList.add('show');
            this.results.scrollIntoView({ behavior: 'smooth' });
        }

        showLoading(show) {
            this.loading.style.display = show ? 'block' : 'none';
            document.getElementById('generateBtn').disabled = show;
            
            if (show) {
                this.results.classList.remove('show');
            }
        }
    }

    function copyCaption(btn, text) {
        navigator.clipboard.writeText(text).then(() => {
            btn.textContent = '‚úÖ Copied!';
            setTimeout(() => {
                btn.textContent = 'üìã Copy Caption';
            }, 2000);
        });
    }

    function showEmailPopup() {
        document.getElementById('emailPopup').classList.add('show');
    }

    function closeEmailPopup() {
        document.getElementById('emailPopup').classList.remove('show');
    }

    document.getElementById('emailCaptureForm').addEventListener('submit', function(e) {
        e.preventDefault();
        alert('üéâ Thank you! Check your email for your free templates. You now have unlimited access!');
        localStorage.setItem('premiumUser', 'true');
        closeEmailPopup();
        
        const generator = window.captionGenerator;
        generator.usageCount = 0;
        generator.maxFreeUses = 999;
        generator.updateUsageDisplay();
    });

    let captionGenerator;
    document.addEventListener('DOMContentLoaded', () => {
        captionGenerator = new CaptionGenerator();
        window.captionGenerator = captionGenerator;
        
        setTimeout(() => {
            if (!localStorage.getItem('emailCaptured') && !localStorage.getItem('premiumUser')) {
                showEmailPopup();
            }
        }, 45000);
    });

    document.getElementById('emailPopup').addEventListener('click', function(e) {
        if (e.target === this) {
            closeEmailPopup();
        }
    });
</script>
